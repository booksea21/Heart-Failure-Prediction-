def or_func(x,y = "DEATH_EVENT"):
    X_v_Y= df[[x,y]].groupby([x,y]).size().reset_index()

    pivot_table = X_v_Y.pivot(index = x, columns = y, values=0)

    res = odds_ratio(pivot_table)
    odds_r = res.statistic
    ci = res.confidence_interval(confidence_level=0.95)

    value_return = {"ods_ratio":odds_r,"lower_ci":ci[0],"upper_ci": ci[1]}
    return value_return

def chisq_test(x,y = "DEATH_EVENT"):
    X_v_Y= df[[x,y]].groupby([x,y]).size().reset_index()

    pivot_table = X_v_Y.pivot(index = x, columns = y, values=0)

    stat, p, dof, expected = chi2_contingency(pivot_table)

    value_return = {"chisqstat":stat,"p-value":p}
    return value_return
